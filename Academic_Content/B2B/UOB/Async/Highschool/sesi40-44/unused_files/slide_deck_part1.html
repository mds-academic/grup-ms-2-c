Di Python, setiap tombol aksi yang dipencet pengguna (Use Case) akan mengaktifkan sebuah fungsi (Function) rahasia.

Dengan cara ini, kode program kita jadi rapi, teratur, dan tidak bertumpuk berantakan!

Fungsi Tambah Jajanan 🛠️

Fungsi ini bertugas menampung catatan baru dari pengguna, memastikan datanya benar (tidak kosong atau salah ketik), lalu memasukannya ke kantong celengan:

def tambah_transaksi():
    nama = input("Nama belanjaan: ").strip()
    jenis = input("Jenis (pemasukan/pengeluaran): ").strip().lower()
    kategori = input("Kategori: ").strip().lower()
    try:
        nominal = int(input("Nominal uang: "))
        if nama == "" or nominal <= 0:
            print("Input salah!")
        else:
            belanja = {"name": nama, "type": jenis, "category": kategori, "amount": nominal}
            transactions.append(belanja)
            print("Tersimpan di celengan!")
    except ValueError:
        print("Nominal uang harus angka ya!")

# Kode Dasar Celengan Pintar
transactions = []

def clean_text(text):
    return text.strip().lower()

def show_menu():
    print("=== SAFE FINANCE TRACKER ===")
    print("1. Tambah transaksi")
    print("2. Ringkasan")
    print("3. Analisis kategori")
    print("4. Rekomendasi")
    print("5. Keluar")

def add_transaction():
    name = input("Nama transaksi: ").strip()
    t_type = clean_text(input("Jenis (pemasukan/pengeluaran): "))
    category = clean_text(input("Kategori: "))
    try:
        amount = int(input("Nominal: "))
        if name == "" or amount <= 0:
            print("Input tidak valid!")
        else:
            transactions.append({
                "name": name, "type": t_type, "category": category, "amount": amount
            })
            print("Berhasil ditambahkan!")
    except ValueError:
        print("Harus angka!")

def summarize_transactions():
    inc = sum(t["amount"] for t in transactions if t["type"] == "pemasukan")
    exp = sum(t["amount"] for t in transactions if t["type"] == "pengeluaran")
    bal = inc - exp
    print("Saldo:", bal)
    return bal

while True:
    show_menu()
    choice = input("Pilihan: ")
    if choice == "1": add_transaction()
    elif choice == "2": summarize_transactions()
    elif choice == "5": break
